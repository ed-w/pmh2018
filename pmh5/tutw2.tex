	\io
	Let $\phi\psi=\psi\phi$ and let $v \in V_\lambda^\phi$, i.e.\@ $\phi v=\lambda v$.
	Then $\phi\psi v = \psi\phi v = \psi\lambda v = \lambda\psi v$ which shows that $\psi[V_\lambda^\phi] \subseteq V_\lambda^\phi$.

	\io
	Define
	\[
		\phi=
		\begin{pmatrix}
			1 & 0 \\
			0 & -1
		\end{pmatrix}
		,\quad

		0 & -1 \\
		1 & 0
		\end{pmatrix}
	\]

		1 & 0 \\

		\end{pmatrix}
	\]

\end{enumerate}
